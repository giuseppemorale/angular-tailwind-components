# Menu

`TailwindMenu` mostra un dropdown con elementi configurabili e trigger per proiezione dei contenuti.

---

## Utilizzo

```typescript
@Component({
  imports: [TailwindMenu, TailwindButton],
  template: `
    <tailwind-menu [items]="items" align="left" (onSelect)="onSelect($event)">
      <tailwind-menu-trigger>
        <tailwind-button>Open menu</tailwind-button>
      </tailwind-menu-trigger>
    </tailwind-menu>`
})
export class ExampleComponent {
  items = [
    { label: 'Profile', value: 'profile' },
    { label: 'Sign out', value: 'signout' }
  ];
  onSelect(item: unknown) {}
}
```
